package inc.nimbuspay.proaccountrest.transaction.entity.objectmodel;

import jakarta.persistence.*;
import java.io.Serializable;
import java.util.List;
import java.util.UUID;

import lombok.*;

@Getter
@Setter
@ToString
@Entity
@Table(name = "transaction", schema = "proaccount")
public class Transaction implements Serializable {
    @Id
    private UUID id;

    @Column(name = "transaction_type")
    private String transactionType;

    @Column(name = "transaction_sub_type")
    private String transactionSubType;

    @Column(name = "transaction_attribute")
    private String transactionAttribute;

    @Column(name = "other_transaction_attribute")
    private String otherTransactionAttribute;

    @ElementCollection
    @CollectionTable(name = "transaction_message_reason", joinColumns = @JoinColumn(name = "transaction_id"), schema = "proaccount")
    @Column(name = "message_reason")
    private List<String> messageReason;

    @ElementCollection
    @CollectionTable(name = "transaction_alternate_message_reason", joinColumns = @JoinColumn(name = "transaction_id"), schema = "proaccount")
    @Column(name = "alternate_message_reason")
    private List<String> alternateMessageReason;

    @Column(name = "pre_authorisation_time_limit")
    private String preAuthorisationTimeLimit;

    @Column(name = "associated_data_indicator")
    private Boolean associatedDataIndicator;

    @Column(name = "associated_data_reference")
    private String associatedDataReference;

    @Column(name = "associated_data_destination")
    private String associatedDataDestination;

    @OneToOne(cascade = CascadeType.ALL)
    @JoinColumn(name = "transaction_identification_id", referencedColumnName = "id")
    private TransactionIdentification transactionIdentification;

    @OneToOne(cascade = CascadeType.ALL)
    @JoinColumn(name = "transaction_amounts_id", referencedColumnName = "id")
    private TransactionAmounts transactionAmounts;

    @Column(name = "transaction_description")
    private String transactionDescription;

    @OneToMany(cascade = CascadeType.ALL)
    @JoinColumn(name = "transaction_id", referencedColumnName = "id")
    private List<AdditionalData> additionalData;

    @OneToOne(cascade = CascadeType.ALL)
    @JoinColumn(name = "account_from_id", referencedColumnName = "id")
    private AccountFrom accountFrom;

    @OneToOne(cascade = CascadeType.ALL)
    @JoinColumn(name = "account_to_id", referencedColumnName = "id")
    private AccountTo accountTo;
}
