package inc.nimbuspay.proaccountrest.transaction.entity.objectmodel;

import jakarta.persistence.*;

import java.io.Serializable;
import java.util.UUID;

import lombok.*;

@Entity
@Getter
@Setter
@ToString
@Table(name = "additional_data", schema = "proaccount")
public class AdditionalData implements Serializable {
    @Id
    @GeneratedValue
    private UUID id;

    @Column(name = "type")
    private String type;

    @Column(name = "value")
    private String value;

}
