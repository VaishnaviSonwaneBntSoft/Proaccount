package inc.nimbuspay.proaccountrest.transaction.entity.objectmodel;

import jakarta.persistence.*;

import java.io.Serializable;
import java.util.UUID;

import lombok.*;

@Getter
@Setter
@ToString
@Entity
@Table(name = "point_of_service_context", schema = "proaccount")
public class PointOfServiceContext implements Serializable {
    @Id
    private UUID id;

    @Column(name = "attended_indicator")
    private Boolean attendedIndicator;

    @Column(name = "card_data_entry_mode")
    private String cardDataEntryMode;

    @Column(name = "is_card_present")
    private Boolean isCardPresent;

    @Column(name = "is_cardholder_activated")
    private Boolean isCardholderActivated;

    @Column(name = "is_cardholder_present")
    private Boolean isCardholderPresent;

    @Column(name = "is_ecommerce")
    private Boolean isEcommerce;

    @Column(name = "is_moto")
    private Boolean isMOTO;

    @Column(name = "is_partial_approval_supported")
    private Boolean isPartialApprovalSupported;

}
