package inc.nimbuspay.proaccountrest.transaction.entity.objectmodel;

import java.io.Serializable;

import jakarta.persistence.Column;
import jakarta.persistence.Entity;
import jakarta.persistence.Id;
import jakarta.persistence.Table;
import java.util.UUID;

import lombok.*;

@Getter
@Setter
@ToString
@Entity
@Table(name = "transaction_context", schema = "proaccount")
public class TransactionContext implements Serializable {
    @Id
    private UUID id;

    @Column(name = "card_programme_app_id")
    private Integer cardProgrammeAppId;

    @Column(name = "is_magnetic_stripe_fallback")
    private Boolean isMagneticStripeFallback;

    @Column(name = "merchant_category_code")
    private Integer merchantCategoryCode;

    @Column(name = "merchant_category_specific_data")
    private String merchantCategorySpecificData;

}
