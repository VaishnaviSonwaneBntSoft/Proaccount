package inc.nimbuspay.proaccountrest.transaction.entity.objectmodel;

import java.io.Serializable;

import jakarta.persistence.Column;
import jakarta.persistence.Entity;
import jakarta.persistence.GeneratedValue;
import jakarta.persistence.Id;
import jakarta.persistence.Table;
import lombok.*;

import java.util.UUID;

@Getter
@Setter
@ToString
@Entity
@Table(name = "transaction_identification", schema = "proaccount")
public class TransactionIdentification implements Serializable {
    @Id
    @GeneratedValue
    private UUID id;

    @Column(name = "local_date")
    private String localDate;

    @Column(name = "system_trace_audit_number")
    private String systemTraceAuditNumber;

    @Column(name = "retrieval_reference_number")
    private String retrievalReferenceNumber;

}
