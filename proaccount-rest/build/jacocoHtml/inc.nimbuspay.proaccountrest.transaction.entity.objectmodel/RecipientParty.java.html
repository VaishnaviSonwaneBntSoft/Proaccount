package inc.nimbuspay.proaccountrest.transaction.entity.objectmodel;

import jakarta.persistence.Column;
import jakarta.persistence.Entity;
import jakarta.persistence.Id;
import jakarta.persistence.Table;
import java.io.Serializable;
import java.util.UUID;

import lombok.*;

@Getter
@Setter
@ToString
@Entity
@Table(name = "recipient_party", schema = "proaccount")
public class RecipientParty implements Serializable {
    @Id
    private UUID id;

    @Column(name = "identification")
    private String identification;

    @Column(name = "type")
    private String type;

    @Column(name = "other_type")
    private String otherType;

    @Column(name = "country")
    private String country;

    @Column(name = "short_name")
    private String shortName;

    @Column(name = "assigner")
    private String assigner;
}
