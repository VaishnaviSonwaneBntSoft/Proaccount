package inc.nimbuspay.proaccountrest.transaction.entity.objectmodel;

import jakarta.persistence.Column;
import jakarta.persistence.Entity;
import jakarta.persistence.Id;
import jakarta.persistence.Table;
import java.io.Serializable;
import java.util.UUID;

import lombok.*;

@Getter
@Setter
@ToString
@Entity
@Table(name = "mac_data", schema = "proaccount")
public class MACData implements Serializable {
    @Id
    private UUID id;

    @Column(name = "control")
    private String control;

    @Column(name = "key_set_identifier")
    private String keySetIdentifier;

    @Column(name = "derived_information")
    private String derivedInformation;

    @Column(name = "algorithm")
    private String algorithm;

    @Column(name = "key_length")
    private String keyLength;

    @Column(name = "key_protection")
    private String keyProtection;

    @Column(name = "key_index")
    private String keyIndex;

    @Column(name = "padding_method")
    private String paddingMethod;

    @Column(name = "initialisation_vector")
    private String initialisationVector;

}
