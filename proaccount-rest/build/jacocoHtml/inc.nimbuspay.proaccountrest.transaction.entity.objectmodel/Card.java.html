package inc.nimbuspay.proaccountrest.transaction.entity.objectmodel;

import jakarta.persistence.*;

import java.io.Serializable;
import java.util.List;
import java.util.UUID;

import lombok.*;

@Getter
@Setter
@ToString
@Entity
@Table(name = "card", schema = "proaccount")
public class Card implements Serializable {
    @Id
    private UUID id;

    @Column(name = "primary_account_number", nullable = false)
    private String primaryAccountNumber;

    @Column(name = "protected_pan_indicator")
    private Boolean protectedPANIndicator;

    @Column(name = "card_sequence_number")
    private String cardSequenceNumber;

    @Column(name = "payment_account_reference")
    private String paymentAccountReference;

    @Column(name = "pan_account_range")
    private String panAccountRange;

    @Column(name = "card_portfolio_identifier")
    private String cardPortfolioIdentifier;

    @Column(name = "card_country_code")
    private String cardCountryCode;

    @OneToMany(cascade = CascadeType.ALL, orphanRemoval = true)
    @JoinColumn(name = "card_id", referencedColumnName = "id")
    private List<AdditionalData> additionalCardData;
}
