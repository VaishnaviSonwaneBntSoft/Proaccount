package inc.nimbuspay.proaccountrest.transaction.entity.objectmodel;

import jakarta.persistence.*;

import java.io.Serializable;
import java.util.UUID;

import lombok.*;

@Entity
@Getter
@Setter
@ToString
@Table(name = "initiating_party", schema = "proaccount")
public class InitiatingParty implements Serializable {
    @Id
    private UUID id;
    @Column(name = "short_name")
    private String shortName;

    @Column(name = "identification")
    private String identification;

    @Column(name = "assigner")
    private String assigner;

    @Column(name = "type")
    private String type;

    @Column(name = "country")
    private String country;

    @Column(name = "other_type")
    private String otherType;

}
