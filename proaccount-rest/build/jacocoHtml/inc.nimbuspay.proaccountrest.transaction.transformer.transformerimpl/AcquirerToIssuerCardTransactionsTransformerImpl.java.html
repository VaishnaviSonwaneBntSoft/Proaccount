package inc.nimbuspay.proaccountrest.transaction.transformer.transformerimpl;

import inc.nimbuspay.proaccountrest.transaction.transformer.AcquirerToIssuerCardTransactionsTransformer;
import org.springframework.stereotype.Service;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import inc.nimbuspay.commonobjectmodel.model.acquirertoissuercardmessages.AcquirerToIssuerCardTransactionsModel;
import inc.nimbuspay.proaccountrest.transaction.mapper.AcquirerToIssuerCardTransactionsMapper;
import inc.nimbuspay.proaccountrest.transaction.request.FinancialInitiation;
import inc.nimbuspay.proaccountrest.transaction.request.ReversalInitiation;
import inc.nimbuspay.proaccountrest.transaction.response.FinancialResponse;
import inc.nimbuspay.proaccountrest.transaction.response.ReversalResponse;
import inc.nimbuspay.proaccountrest.transaction.mapper.AcquirerToIssuerCardTransactionMapperForReversal;

@Service
public class AcquirerToIssuerCardTransactionsTransformerImpl
        implements AcquirerToIssuerCardTransactionsTransformer {

    private static final Logger logger =
            LoggerFactory.getLogger(AcquirerToIssuerCardTransactionsTransformerImpl.class);

    @Override
    public AcquirerToIssuerCardTransactionsModel transformFinancialInitiation(
            FinancialInitiation financialInitiation) {

        AcquirerToIssuerCardTransactionsModel acquirerToIssuerCardTransactionsModel =
                AcquirerToIssuerCardTransactionsMapper
                        .INSTANCE
                        .financialInitiationToAcquirerToIssuerCardTransactionsModel(financialInitiation);

        logger.info("AcquirerToIssuerCardTransactionsModel :{}", acquirerToIssuerCardTransactionsModel);
        return acquirerToIssuerCardTransactionsModel;
    }

    @Override
    public FinancialResponse transformFinancialResponse(
            AcquirerToIssuerCardTransactionsModel acquirerToIssuerCardTransactionsModel) {

        return AcquirerToIssuerCardTransactionsMapper
                        .INSTANCE
                        .acquirerToIssuerCardTransactionsModelTofinancialResponse(acquirerToIssuerCardTransactionsModel);
    }


     @Override
        public AcquirerToIssuerCardTransactionsModel transformReversalInitiation(
                        ReversalInitiation riversalInitiation) {
                return AcquirerToIssuerCardTransactionMapperForReversal.INSTANCE
                                .reversalInitiationToAcquirerToIssuerCardTransactionsModel(riversalInitiation);
        } 

        public ReversalResponse transformReversalResponse(
                        AcquirerToIssuerCardTransactionsModel acquirerToIssuerCardTransactionsModel) {
                                return AcquirerToIssuerCardTransactionMapperForReversal.INSTANCE
                                .acquirerToIssuerCardTransactionsModelToReversalResponse(
                                                acquirerToIssuerCardTransactionsModel);
        }
}
