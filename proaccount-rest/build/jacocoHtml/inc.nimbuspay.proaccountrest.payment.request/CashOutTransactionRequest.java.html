package inc.nimbuspay.proaccountrest.payment.request;

import jakarta.validation.constraints.Digits;
import jakarta.validation.constraints.Min;
import jakarta.validation.constraints.NotBlank;
import jakarta.validation.constraints.NotNull;
import jakarta.validation.constraints.Pattern;
import lombok.AllArgsConstructor;
import lombok.Builder;
import lombok.Data;
import lombok.NoArgsConstructor;

@Data
@AllArgsConstructor
@NoArgsConstructor
@Builder
public class CashOutTransactionRequest {

    @NotNull(message = "Debit Account Number can not be null")
    @Digits(integer = 12, fraction = 0, message = "Debit Account Number must be exactly 12 digits long")
    private Long sourceAccount;

    @NotBlank(message = "TransactionType can not be empty")
    @NotNull(message = "TransactionType can not be null")
    private String transactionType;

    @Min(value = 1, message = "Amount must be greater than 0")
    private Double transactionValue;

    @Pattern(regexp = "^[A-Z]{3}$", message = "Currency code must be 3 uppercase letters")
    private String iso4217Currency;

    @NotNull(message = "credit Account Number can not be null")
    @Digits(integer = 12, fraction = 0, message = "Credit Account Number must be exactly 12 digits long")
    private Long destAccountNumber;

    private String destAccountSrtCde;

    private String destAccIban;

    @NotNull
    private String destAccountName;

    @NotNull
    private String transactionSource;

    private String payRef;

}
