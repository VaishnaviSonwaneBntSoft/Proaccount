package inc.nimbuspay.proaccountrest.exceptions;

import org.springframework.http.HttpStatusCode;

import lombok.Getter;
import lombok.ToString;

@Getter
@ToString
public class PaymentException extends RuntimeException{

    private final String exceptionMessage;

    private final Integer statusCode;

    private final HttpStatusCode status;

    private final String workFlow;

    private  String exception;

    public PaymentException(String message, Integer statusCode, HttpStatusCode status,String workFlow) {
        super(message);
        this.exceptionMessage = message;
        this.statusCode = statusCode;
        this.status = status;
        this.workFlow=workFlow;
    }

    public PaymentException(String message, Integer statusCode, HttpStatusCode status,String workFlow,String exception) {
        super(message);
        this.exceptionMessage = message;
        this.statusCode = statusCode;
        this.status = status;
        this.workFlow=workFlow;
        this.exception=exception;
    }
}
