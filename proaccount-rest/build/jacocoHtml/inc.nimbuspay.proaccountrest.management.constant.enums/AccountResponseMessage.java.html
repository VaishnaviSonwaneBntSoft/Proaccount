package inc.nimbuspay.proaccountrest.management.constant.enums;

public enum AccountResponseMessage {
    SUCCESSFULLY_DELETED("Account with account number %s deleted successfully"),
    SUCCESSFULLY_UPDATED("Account with account number %s updated successfully"),
    ACCOUNT_ALREADY_DELETED("Account with account number %s already deleted"),
    ACCOUNT_ALREADY_EXISTS("Account number already exists"),
    ACCOUNT_NOT_FOUND("Account with  accountNumber %s is not found."),
    ACCOUNT_BALANCE_MUST_BE_ZERO("Please clear funds before In-activating Account"),
    FAILED_TO_SAVE_ACCOUNT("Failed to save account");

    private final String messageFormat;

    AccountResponseMessage(String messageFormat) {
        this.messageFormat = messageFormat;

    }

    public String getMessage(Long accountNumber) {
        return String.format(messageFormat, accountNumber);
    }

    public String getMessage(String accountNumber) {
        return String.format(messageFormat, accountNumber);
    }

    public String getMessage() {
        return messageFormat;
    }

}
