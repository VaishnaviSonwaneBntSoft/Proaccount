package inc.nimbuspay.proaccountrest.payment.constant.enums;

public enum PaymentResponseMessage {

    ACCOUNT_NOT_FOUND("Account with  accountNumber %s is not found."),
    FAILED_FETCH_ACCOUNT("Failed to fetch account by type"),
    FAILED_SAVE_ACC_REC_LOG("Failed to save AccountReceivableLog"),
    FAILED_SAVE_ACC_PAY_LOG("Failed to save AccountPayableLog"),
    VIRTUAL_ACCOUNT_NOT_FOUND("Account not found"),
    ACCOUNT_NOT_FOUND_WITH_PAY_REF_NUMBER("Account with  payment Reference %s is not found."),
    TRANSACTION_FAIL("Transaction is failed"),
    TRANSACTION_SUCCESSFUL("Transaction processed successfully"),
    ACCOUNT_BALANCE_MUST_BE_ZERO("Please clear funds before In-activating Account"),
    ACCOUNT_BALANCE_SHOULD_NOT_LESSER_THAN_DEBIT_AMOUNT("Insufficient funds"),
    ACCOUNT_BALANCE_UNAVAILABLE("Debtor account has 0 balance"),
    GENERATED_UNIQUE_PAYMENT_REFERENCE("PaymentReference : %s"),
    PAYMENT_REFERENCE_ALREADY_EXIST("Invalid Payment Reference,Please Genereate New Payment Reference"),
    PAYMENT_REFERENCE_NOT_FOUND("Payment Reference not found."),
    INVALID_CURRENCY("Invalid Currency."),
    DELETED_ACCOUNT("%s account is mark deleted");

    private final String messageFormat;

    PaymentResponseMessage(String messageFormat) {
        this.messageFormat = messageFormat;

    }

    public String getMessage(Long accountNumber) {
        return String.format(messageFormat, accountNumber);
    }

    public String getMessage(String accountNumber) {
        return String.format(messageFormat, accountNumber);
    }

    public String getMessage() {
        return messageFormat;
    }
}
