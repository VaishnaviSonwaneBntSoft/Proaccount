package inc.nimbuspay.proaccountrest.transaction.controller;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;

import inc.nimbuspay.proaccountrest.transaction.request.FinancialInitiation;
import inc.nimbuspay.proaccountrest.transaction.request.ReversalInitiation;
import inc.nimbuspay.proaccountrest.transaction.response.FinancialResponse;
import inc.nimbuspay.proaccountrest.transaction.response.ReversalResponse;
import inc.nimbuspay.proaccountrest.transaction.service.AcquirerToIssuerCardTransactionsService;

@RestController
@RequestMapping("/api/transaction/acquirer-to-issuer-card-transactions")
public class AcquirerToIssuerCardTransactionsController {
    private static final Logger logger = LoggerFactory
            .getLogger(AcquirerToIssuerCardTransactionsController.class);

    private AcquirerToIssuerCardTransactionsService acquirerToIssuerCardTransactionsService;

    public AcquirerToIssuerCardTransactionsController(
            AcquirerToIssuerCardTransactionsService acquirerToIssuerCardTransactionsService) {
        this.acquirerToIssuerCardTransactionsService = acquirerToIssuerCardTransactionsService;
    }

    @PostMapping("/financial")
    public FinancialResponse financial(
            @RequestBody FinancialInitiation financialInitiation) {

        logger.info("Financial Initiation Message : {}", financialInitiation);

        FinancialResponse financialResponse = acquirerToIssuerCardTransactionsService.financial(financialInitiation);

        logger.info("Financial Response Message : {}", financialResponse);
        return financialResponse;

    }

    @PostMapping("/reversal")
    public ReversalResponse reversal(@RequestBody ReversalInitiation reversalInitiation) {
        logger.info("Reversal Initiation Message : {}", reversalInitiation);
        ReversalResponse reversalResponce = acquirerToIssuerCardTransactionsService.reversal(reversalInitiation);
        logger.info("Reversal Response Message : {}", reversalResponce);
        return reversalResponce;
    }

}
